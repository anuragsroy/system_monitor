pipeline {
    agent any

    stages {
	stage('minikube') {
            steps {
                bat 'minikube start'
            }
        }
        stage('git import') {
            steps {
                git branch: 'main', url: 'https://github.com/anuragsroy/yaml-files-project.git'
            }
        }
        stage('pod creation') {
            steps {
                bat 'kubectl apply -f project_manifest_backend.yml'
            }
        }
        stage('configmap creation') {
            steps {
                bat 'kubectl apply -f project_configmap_backend.yml'
            }
        }
        stage('service creation') {
            steps {
                bat 'kubectl apply -f project_service_backend.yml'
            }
        }
        stage('deployment creation') {
            steps {
                bat 'kubectl apply -f project_deployment_backend.yml'
            }
        }
    }
}
